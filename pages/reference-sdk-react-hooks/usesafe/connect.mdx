import { Tabs } from 'nextra/components'

# `connect`

Connects the specified signer.

## Usage

{/* <!-- vale off --> */}

<Tabs items={['example.tsx']}>
  <Tabs.Tab>
    ```typescript
    import { useSafe } from '@safe-global/safe-core-sdk-react-hooks'

    const Connect = () => {
      const { connect } = useSafe()

      const signerAddress = '0x...'

      return (
        <>
          <h3>Chain Info</h3>
          <button onClick={() => connect(signerAddress)}></button>
        </>
      )
    }

    export default Connect
    ```
  </Tabs.Tab>
</Tabs>

{/* <!-- vale on --> */}

## Returns

`(signer: string) => Promise<void>`

## Parameters

### `signer`

- **Type**: `string`

The address of the signer to connect.
